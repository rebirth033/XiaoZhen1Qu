-- Create table
create table CODES_CL
(
  codeid    NUMBER(12) not null,
  typename  VARCHAR2(60),
  codename  VARCHAR2(60),
  codevalue VARCHAR2(120),
  codeorder NUMBER(3),
  parentid  NUMBER(12)
)
tablespace USERS
  pctfree 10
  initrans 1
  maxtrans 255
  storage
  (
    initial 16K
    next 1M
    minextents 1
    maxextents unlimited
  );
-- Add comments to the columns 
comment on column CODES_CL.codeid
  is '主键';
comment on column CODES_CL.typename
  is '类型';
comment on column CODES_CL.codename
  is '名称';
comment on column CODES_CL.codevalue
  is '值';
comment on column CODES_CL.codeorder
  is '排序';
comment on column CODES_CL.parentid
  is '父级ID';
