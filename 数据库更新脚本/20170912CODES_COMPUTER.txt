-- Create table
create table CODES_COMPUTER
(
  codeid    NUMBER(12) not null,
  typename  VARCHAR2(60),
  codename  VARCHAR2(60),
  codevalue VARCHAR2(120),
  codeorder NUMBER(3),
  parentid  NUMBER(12)
)
tablespace USERS
  pctfree 10
  initrans 1
  maxtrans 255
  storage
  (
    initial 16
    minextents 1
    maxextents unlimited
  );
-- Add comments to the table 
comment on table CODES_COMPUTER
  is '码表';
-- Add comments to the columns 
comment on column CODES_COMPUTER.codeid
  is '主键';
comment on column CODES_COMPUTER.typename
  is '类型';
comment on column CODES_COMPUTER.codename
  is '名称';
comment on column CODES_COMPUTER.codevalue
  is '值';
comment on column CODES_COMPUTER.codeorder
  is '排序';
comment on column CODES_COMPUTER.parentid
  is '父级ID';
-- Create/Recreate primary, unique and foreign key constraints 
alter table CODES_COMPUTER
  add constraint PK_CODES_COMPUTER primary key (CODEID)
  using index
