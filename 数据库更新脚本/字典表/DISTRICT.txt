-- Create table
create table DISTRICT
(
  name      VARCHAR2(100),
  supername VARCHAR2(100),
  code      VARCHAR2(20) not null,
  supercode VARCHAR2(20),
  grade     VARCHAR2(10),
  adlevel   NUMBER(1),
  area      VARCHAR2(10),
  shortname VARCHAR2(10)
)
tablespace USERS
  pctfree 10
  initrans 1
  maxtrans 255
  storage
  (
    initial 3
    minextents 1
    maxextents unlimited
  );
-- Add comments to the table 
comment on table DISTRICT

  is '父级行政区名称';

  is '行政区编码';
comment on column DISTRICT.supercode
  is '父级行政区编码';

  is '级别';
