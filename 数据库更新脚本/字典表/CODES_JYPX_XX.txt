-- Create table
create table CODES_JYPX_XX
(
  codeid    NUMBER(12) not null,
  typename  VARCHAR2(60),
  codename  VARCHAR2(60),
  codevalue VARCHAR2(120),
  codeorder NUMBER(3),
  parentid  NUMBER(12)
)
tablespace USERS
  pctfree 10
  initrans 1
  maxtrans 255
  storage
  (
    initial 64
    minextents 1
    maxextents unlimited
  );

  is '类型';

  is '名称';

  is '值';

  is '排序';
