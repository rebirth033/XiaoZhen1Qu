CREATE TABLE `CODES_DISTRICT` (
  `codeid` INT(12) NOT NULL,
  `typename` VARCHAR(60) DEFAULT NULL,
  `codename` VARCHAR(60) DEFAULT NULL,
  `codevalue` VARCHAR(60) DEFAULT NULL,
  `codeorder` INT(10) DEFAULT NULL,
  `parentid` INT(12) DEFAULT NULL,
  `shortname` VARCHAR(10) DEFAULT NULL,
  PRIMARY KEY (`codeid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
